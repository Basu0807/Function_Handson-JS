Q-7 Write a function that accepts parameter n and returns factorial of n

function factorial(n) {
            if (n === 0 || n === 1) {
              return 1;
            } else {
              let result = 1;
              for (let i = 2; i<= n; i++) {
                result *= i;
              }
              return result;
            }
          }
